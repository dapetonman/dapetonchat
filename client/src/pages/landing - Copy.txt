import { useLocation } from "wouter";

function TypeText({ text, speed = 50, onComplete, className }: { text: string; speed?: number; onComplete?: () => void; className?: string }) {
  const [displayed, setDisplayed] = useState("");

  useEffect(() => {
    let i = 0;
    setDisplayed("");
    const interval = setInterval(() => {
      if (i < text.length) {
        setDisplayed(text.slice(0, i + 1));
        i++;
      } else {
        clearInterval(interval);
        onComplete?.();
      }
    }, speed);
    return () => clearInterval(interval);
  }, [text, speed, onComplete]);

  return <span className={className}>{displayed}</span>;
}

function TypedButton({ text, index, shouldReveal, onTyped, onClick }: { text: string; index: number; shouldReveal: boolean; onTyped: () => void; onClick: () => void }) {
  const [displayed, setDisplayed] = useState("");
  const [started, setStarted] = useState(false);

  useEffect(() => {
    if (!shouldReveal || started) return;
    setStarted(true);
    let i = 0;
    const timeout = setTimeout(() => {
      const interval = setInterval(() => {
        if (i < text.length) {
          setDisplayed(text.slice(0, i + 1));
          i++;
        } else {
          clearInterval(interval);
          onTyped();
        }
      }, 20);
      return () => clearInterval(interval);
    }, index === 0 ? 0 : 500);
    return () => clearTimeout(timeout);
  }, [shouldReveal, started, text, index, onTyped]);

  if (!started && !shouldReveal) return null;

  return (
    <button className="block w-full border-2 border-white text-white text-3xl py-2.5 bg-transparent hover:bg-white hover:text-black cursor-pointer transition-[0.1s] mb-2.5" style={{ fontFamily: "'VT323', monospace" }} onClick={onClick}>
      {displayed}
    </button>
  );
}

function BackButton({ onClick }: { onClick: () => void }) {
  return (
    <button className="block w-full border-2 border-white text-white text-3xl py-2.5 bg-transparent hover:bg-white hover:text-black cursor-pointer transition-[0.1s]" style={{ fontFamily: "'VT323', monospace" }} onClick={onClick}>[ BACK ]</button>
  );
}
